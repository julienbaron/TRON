3   import java.awt.event.KeyEvent;
4   
5   import controller.IOrderPerformer;
6   import controller.IUserOrder;
7   import controller.Order;
8   import controller.UserOrder;
9   import gameframe.IEventPerformer;
10  
11  class EventPerformer implements IEventPerformer {
12  	private final IOrderPerformer orderPerformer;
13  
14  	public EventPerformer(final IOrderPerformer orderPerformer) {
15  		this.orderPerformer = orderPerformer;
16  	}
17  
18  	@Override
19  	public void eventPerform(final KeyEvent keyCode) {
20  		final IUserOrder userOrder = this.keyCodeToUserOrder(keyCode.getKeyCode());
21  		if (userOrder != null) {
22  			this.orderPerformer.orderPerform(userOrder);
23  		}
24  	}
25  
26  	public IUserOrder keyCodeToUserOrder(final int keyCode) {
27  		IUserOrder userOrder;
28  		switch (keyCode) {
29  			case KeyEvent.VK_UP:
30  				userOrder = new UserOrder(0, Order.UP);
31  				break;
32  			case KeyEvent.VK_RIGHT:
33  				userOrder = new UserOrder(0, Order.RIGHT);
34  				break;
35  			case KeyEvent.VK_DOWN:
36  				userOrder = new UserOrder(0, Order.DOWN);
37  				break;
38  			case KeyEvent.VK_LEFT:
39  				userOrder = new UserOrder(0, Order.LEFT);
40  				break;
41  			case KeyEvent.VK_CONTROL:
42  				userOrder = new UserOrder(0, Order.SHOOT);
43  				break;
44  
45  			case KeyEvent.VK_Z:
46  				userOrder = new UserOrder(1, Order.UP);
47  				break;
48  			case KeyEvent.VK_D:
49  				userOrder = new UserOrder(1, Order.RIGHT);
50  				break;
51  			case KeyEvent.VK_S:
52  				userOrder = new UserOrder(1, Order.DOWN);
53  				break;
54  			case KeyEvent.VK_Q:
55  				userOrder = new UserOrder(1, Order.LEFT);
56  				break;
57  			case KeyEvent.VK_SHIFT:
58  				userOrder = new UserOrder(1, Order.SHOOT);
59  				break;
60  			default:
61  				//userOrder = new UserOrder(0, Order.NOP);
62  				//System.out.println("nop2");
63  				userOrder = null;
64  		}
65  	
66  		return userOrder;
67  	}
68  }
